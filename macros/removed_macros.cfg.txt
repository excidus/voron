[gcode_macro PRIME_NOZZLE]
gcode:
    SAVE_GCODE_STATE NAME=PRIME_NOZZLE_STATE 
    G90 ; Ensure absolute positioning
    G92 E0 ; Reset extruder position
    G1 X4 Y4 F9000
    G1 Y44 E1 F900
    RESTORE_GCODE_STATE NAME=PRIME_NOZZLE_STATE
